function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // Create output let result = { status: "success" }; try { const data = JSON.parse(e.postData.contents); const date = data.date; const member = data.member; const tasks = data.tasks; // Array of {time, task, status, notes} if (!date || !member || !tasks) { throw new Error("Missing required parameters: date, member, tasks"); } // Handle global broadcast message update const broadcast = data.broadcast; if (broadcast !== undefined) { PropertiesService.getScriptProperties().setProperty('broadcast', broadcast); } // 1. Find and delete existing rows for this date + member combination const dataRange = sheet.getDataRange(); const values = dataRange.getValues(); // Use raw values instead of formatted const rowsToDelete = []; // Skip header row (i = 0) for (let i = 1; i < values.length; i++) { let rDate = values[i][0]; // Force Google Sheet date objects or strings into strict YYYY-MM-DD if (rDate instanceof Date) { const y = rDate.getFullYear(); const m = String(rDate.getMonth() + 1).padStart(2, '0'); const d = String(rDate.getDate()).padStart(2, '0'); rDate = `${y}-${m}-${d}`; } else { rDate = String(rDate).replace(/^'/, '').trim(); } let rMember = String(values[i][1]).replace(/^'/, '').trim(); if (rDate === date && rMember === member) { rowsToDelete.push(i + 1); // +1 because row index is 1-based in Sheets } } // Delete from bottom up to avoid shifting index issues for (let i = rowsToDelete.length - 1; i >= 0; i--) { sheet.deleteRow(rowsToDelete[i]); } // 2. Insert the new tasks, prefixing date and time with ' so Sheets treats them as pure text // Expected Columns: A=Date, B=Member, C=Time, D=Task, E=Type, F=Status, G=Notes const newRows = tasks.map(t => ["'" + date, member, "'" + t.time, t.task, t.type || 'empty', t.status || 'empty', t.notes || '']); if (newRows.length > 0) { sheet.getRange(sheet.getLastRow() + 1, 1, newRows.length, 7).setValues(newRows); } } catch (error) { result = { status: "error", message: error.toString() }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); let result = { status: "success", data: [], broadcast: "" }; try { result.broadcast = PropertiesService.getScriptProperties().getProperty('broadcast') || ''; const date = e.parameter.date; const member = e.parameter.member; if (!date) { throw new Error("Missing required parameter: date"); } const dataRange = sheet.getDataRange(); const values = dataRange.getValues(); // Expected Columns: A=Date, B=Member, C=Time, D=Task, E=Type, F=Status, G=Notes // Start i = 1 to skip header row for (let i = 1; i < values.length; i++) { let rDate = values[i][0]; // Force Google Sheet date objects or strings into strict YYYY-MM-DD if (rDate instanceof Date) { const y = rDate.getFullYear(); const m = String(rDate.getMonth() + 1).padStart(2, '0'); const d = String(rDate.getDate()).padStart(2, '0'); rDate = `${y}-${m}-${d}`; } else { rDate = String(rDate).replace(/^'/, '').trim(); } let rMember = String(values[i][1]).replace(/^'/, '').trim(); if (rDate === date && (!member || rMember === member)) { let rTime = values[i][2]; if (rTime instanceof Date) { const hours = String(rTime.getHours()).padStart(2, '0'); const minutes = String(rTime.getMinutes()).padStart(2, '0'); rTime = `${hours}:${minutes}`; } else { rTime = String(rTime).replace(/^'/, '').trim(); } result.data.push({ member: rMember, time: rTime, task: values[i][3], type: values[i][4], status: values[i][5], notes: values[i][6] }); } } } catch (error) { result = { status: "error", message: error.toString() }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doOptions(e) { return ContentService.createTextOutput() .setMimeType(ContentService.MimeType.TEXT) .setContent(''); } // No replacement, deleting duplicate doOptions // Setup function to run once to create headers function setupSheet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(["Date", "Member", "Time", "Task", "Type", "Status", "Notes"]); sheet.getRange("A1:G1").setFontWeight("bold"); sheet.setFrozenRows(1); } }
